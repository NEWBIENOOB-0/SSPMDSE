void InsertBeg()
{
	node = (emp*) malloc(1* sizeof(emp));
	node->next = NULL;
	node->prev = NULL;
	printf("\nenter info: ");
	scanf("%d", &node->info);
	if (list == NULL)
	{
		list = node;
	}
	else
	{
		list->prev = node;
		node->next = list;
		list = node;
	}
}

void DeleteBeg()
{
	if (list == NULL)
	{
		printf("\nLIST IS AN EMPTY....");
	}
	else
	{
		node = list;
		printf("\nthe element deleted is: %d", list->info);
		list = list->next;
		if (list != NULL)
			list->prev = NULL;
		free(node);
	}
}

void InsertEnd()
{
	node = (emp*) malloc(1* sizeof(emp));
	node->next = NULL;
	node->prev = NULL;
	printf("\nenter info: ");
	scanf("%d", &node->info);
	temp = list;
	if (temp == NULL)
	{
		list = node;
	}
	else
	{
		while (temp->next != NULL)
		{
			temp = temp->next;
		}

		temp->next = node;
		node->prev = temp;
	}
}

void DeleteEnd()
{
	temp = list;
	if (temp == NULL)
	{
		printf("\nLIST IS EMPTY...");
	}
	else
	{
		while (temp->next != NULL)
		{
			temp = temp->next;
		}

		if (temp->prev != NULL)
		{
			printf("\nelement deleted is: %d", temp->info);
			(temp->prev)->next = NULL;
		}
		else
		{
			printf("\nelement deleted is: %d", list->info);
			list = NULL;
		}
	}
}

void Display()
{
	temp = list;
	if (temp == NULL)
	{
		printf("\nEMPTY LINKED LIST...");
	}
	else
	{
		printf("\nLIST INCLUDES: ");
		while (temp != NULL)
		{
			printf(" %d", temp->info);
			temp = temp->next;
		}
	}
}
